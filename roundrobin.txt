#include <stdio.h>

int main() {
    int n, i, tq, remaining;
    float avg_wait = 0, avg_tat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int bt[n], rt[n], wt[n], tat[n];
    int time = 0;

    // Input burst times
    printf("Enter burst time for each process:\n");
    for (i = 0; i < n; i++) {
        printf("P%d: ", i + 1);
        scanf("%d", &bt[i]);
        rt[i] = bt[i]; // remaining time
    }

    // Input time quantum
    printf("Enter time quantum: ");
    scanf("%d", &tq);

    remaining = n;

    // Round Robin Scheduling
    while (remaining != 0) {
        int done = 1; // assume all done
        for (i = 0; i < n; i++) {
            if (rt[i] > 0) {
                done = 0; // found unfinished process
                if (rt[i] > tq) {
                    time += tq;
                    rt[i] -= tq;
                } else {
                    time += rt[i];
                    wt[i] = time - bt[i];
                    rt[i] = 0;
                    remaining--;
                }
            }
        }
        if (done == 1)
            break;
    }

    // Calculate Turnaround Time
    for (i = 0; i < n; i++) {
        tat[i] = bt[i] + wt[i];
        avg_wait += wt[i];
        avg_tat += tat[i];
    }

    avg_wait /= n;
    avg_tat /= n;

    // Display results
    printf("\nProcess\tBurst Time\tWaiting Time\tTurnaround Time\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t\t%d\t\t%d\t\t%d\n", i + 1, bt[i], wt[i], tat[i]);
    }

    printf("\nAverage Waiting Time: %.2f", avg_wait);
    printf("\nAverage Turnaround Time: %.2f\n", avg_tat);

    return 0;
}
